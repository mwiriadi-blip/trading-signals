```python
def _fmt_em_dash() -> str:

def _fmt_currency(value: float) -> str:
  if value < 0:
    return f'-${-value:,.2f}'
  return f'${value:,.2f}'

def _fmt_percent_signed(fraction: float) -> str:
  return f'{fraction * 100:+.1f}%'

def _fmt_percent_unsigned(fraction: float) -> str:
  return f'{fraction * 100:.1f}%'

def _fmt_last_updated(now: datetime) -> str:
  # Naive datetime rejection (C-1 reviews)
  if now.tzinfo is None:
    raise ValueError(
      '_fmt_last_updated_email requires a timezone-aware datetime; '
      f'got naive datetime={now!r}'
    )
  awst = now.astimezone(pytz.timezone('Australia/Perth'))
  return awst.strftime('%Y-%m-%d %H:%M AWST')
```

D-06 signal-change detection (CONTEXT line 114-116):
```python
def _detect_signal_changes(state: dict, old_signals: dict) -> bool:
  '''Returns True iff any instrument's signal changed from old_signals.
  First-run / no-previous case (old is None) treated as NO CHANGE per D-06.
  '''
  for yf_sym, state_key in (('^AXJO', 'SPI200'), ('AUDUSD=X', 'AUDUSD')):
    old = old_signals.get(yf_sym)
    raw = state['signals'].get(state_key)
    new = raw if isinstance(raw, int) else raw.get('signal', 0)
    if old is not None and old != new:
      return True
  return False
```

    ```python
    def _closed_position_for_instrument_on(
      state: dict, state_key: str, run_date_iso: str,
    ) -> dict | None:
      '''UI-SPEC §2 Option A: return the most recent trade_log entry for
      `state_key` closed on `run_date_iso`, scanning up to the last 3 records.

      Returns the trade record dict (12 fields) for the most recent matching
      trade, where matching = exit_date == run_date_iso AND instrument ==
      state_key. Returns None if no match within the last 3 records.

      Widened-scan rationale (REVIEWS.md Fix 4): if BOTH instruments close
      on the same run (SPI200 AND AUDUSD reversal on same day),
      trade_log[-1] only covers one instrument. Scanning the last 3 records
      (tail + 2 preceding) covers same-run double-close without risk of
      picking up stale matches from earlier days.

      appended to trade_log during run_daily_check. trade_log tail is
    ```
