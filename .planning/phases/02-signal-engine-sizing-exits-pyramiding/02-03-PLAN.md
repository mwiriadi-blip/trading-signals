def get_trailing_stop(position: Position, current_price: float, atr: float) -> float:
  '''EXIT-06/07: compute current trailing stop price.

  LONG:  stop = peak_price - TRAIL_MULT_LONG * atr   (peak_price is highest HIGH since entry)
  SHORT: stop = trough_price + TRAIL_MULT_SHORT * atr (trough_price is lowest LOW since entry)

  Pitfall 3: peak_price (LONG) or trough_price (SHORT) may be None on the first
  bar of a new position before peak/trough has been initialized by the orchestrator.
  In that case fall back to entry_price — the conservative choice that gives a
  stop equal to entry +/- trail_mult*atr (no profit lock yet).

  CLAUDE.md operator decision: intraday HIGH/LOW drives BOTH peak/trough updates
  AND hit detection. The peak/trough fields stored on Position must already
  reflect today's HIGH or LOW (the orchestrator updates them BEFORE calling this).

  current_price parameter is reserved for future "force initialise from current"
  callers; today the function only needs peak/trough/entry. Kept in signature for
  API stability — adding it later would break Phase 4 callers.
  '''
  del current_price  # Reserved; intentional unused parameter (kept for API stability).
  if position['direction'] == 'LONG':
    peak = position['peak_price']
    if peak is None:
      peak = position['entry_price']
    return peak - TRAIL_MULT_LONG * atr
  # SHORT branch
  trough = position['trough_price']
  if trough is None:
    trough = position['entry_price']
  return trough + TRAIL_MULT_SHORT * atr
```

def check_stop_hit(position: Position, high: float, low: float, atr: float) -> bool:
  '''EXIT-08/09: True if today's intraday bar hit the trailing stop.

  LONG:  hit if low  <= stop, where stop = peak_price - TRAIL_MULT_LONG * atr
  SHORT: hit if high >= stop, where stop = trough_price + TRAIL_MULT_SHORT * atr

  Boundary: <= and >= are INCLUSIVE — a low or high exactly at the stop is a hit.

  CLAUDE.md operator decision: intraday HIGH/LOW (NOT close-only). Phase 2 fixtures
  must supply both HIGH and LOW; close-only fixtures are under-specified for this phase.

  Pitfall 2: this function returns BOOL ONLY. Fill price logic (stop vs gap-open
  for gap-through-stop case) is Phase 3 record_trade territory.

  Pitfall 3: peak_price/trough_price None safety mirrors get_trailing_stop.
  '''
  if position['direction'] == 'LONG':
    peak = position['peak_price']
    if peak is None:
      peak = position['entry_price']
    stop = peak - TRAIL_MULT_LONG * atr
    return low <= stop
  # SHORT branch
  trough = position['trough_price']
  if trough is None:
    trough = position['entry_price']
  stop = trough + TRAIL_MULT_SHORT * atr
  return high >= stop
```
