```python
'''System parameters — policy constants and shared types for the signal/sizing hex.

Holds all trading-policy constants (indicator periods, ADX/Mom thresholds, sizing risk
percentages, vol-scale clip range, pyramid thresholds, contract specs) and the
`Position` TypedDict consumed by sizing_engine.py and persisted by Phase 3
state_manager.py. Phase 2 D-01 moved Phase 1 policy constants out of signal_engine.py
into this module so signal_engine.py can stay focused on indicator math + vote.

Architecture (hexagonal-lite, CLAUDE.md): pure constants + types ONLY. No I/O,
no network, no math functions, no imports of state_manager / notifier / dashboard /
sizing_engine / signal_engine / main. The only allowed import is `typing` (TypedDict, Literal).
'''
from typing import Literal, TypedDict


# --- Indicator periods (Phase 1, migrated per D-01) ---
ATR_PERIOD: int = 14
ADX_PERIOD: int = 20
MOM_PERIODS: tuple[int, int, int] = (21, 63, 252)
RVOL_PERIOD: int = 20
ANNUALISATION_FACTOR: int = 252

# --- Vote thresholds (Phase 1, migrated per D-01) ---
ADX_GATE: float = 25.0
MOM_THRESHOLD: float = 0.02

# --- Sizing parameters (Phase 2, SIZE-01..04) ---
RISK_PCT_LONG: float = 0.01     # SIZE-01
RISK_PCT_SHORT: float = 0.005   # SIZE-01
TRAIL_MULT_LONG: float = 3.0    # SIZE-02 / EXIT-06
TRAIL_MULT_SHORT: float = 2.0   # SIZE-02 / EXIT-07
VOL_SCALE_TARGET: float = 0.12  # SIZE-03 numerator
VOL_SCALE_MIN: float = 0.3      # SIZE-03 floor
VOL_SCALE_MAX: float = 2.0      # SIZE-03 ceiling (also RVol NaN/zero fallback)

# --- Pyramid parameters (Phase 2, PYRA-02/03/04) ---
PYRAMID_TRIGGERS: tuple[float, float] = (1.0, 2.0)  # multiples of atr_entry; level 0->1 at 1x, level 1->2 at 2x
MAX_PYRAMID_LEVEL: int = 2      # PYRA-04: cap at 3 total contracts (level 0=1, 1=2, 2=3)

# --- Exit gate (Phase 2, EXIT-05) ---
ADX_EXIT_GATE: float = 20.0     # close position immediately if today's ADX falls below this

# --- Contract specs (Phase 2, D-11 — operator confirmed at /gsd-discuss-phase 2) ---
SPI_MULT: float = 5.0           # SPI 200 mini, AUD per index point
SPI_COST_AUD: float = 6.0       # round-trip ($3 open + $3 close per D-13)
AUDUSD_NOTIONAL: float = 10000.0  # mini-lot notional in USD-per-AUD terms
AUDUSD_COST_AUD: float = 5.0    # round-trip ($2.50 open + $2.50 close per D-13)


# --- Position TypedDict (Phase 2, D-08) ---
# Consumed by sizing_engine.py functions; persisted by Phase 3 state_manager.py.
# JSON-round-trip friendly (TypedDict IS a dict at runtime).
class Position(TypedDict):
  direction: Literal['LONG', 'SHORT']
  entry_price: float
  entry_date: str            # ISO YYYY-MM-DD
  n_contracts: int
  pyramid_level: int         # 0, 1, or 2 (capped by MAX_PYRAMID_LEVEL)
  peak_price: float | None   # LONG: highest HIGH since entry; None for SHORT
  trough_price: float | None # SHORT: lowest LOW since entry; None for LONG
  atr_entry: float           # ATR at entry time, used for pyramid thresholds
```

```python
'''Sizing Engine — pure-math position sizing, trailing stops, pyramid state machine.

Phase 2 D-07 module. Consumes scalars (atr, rvol, prices) and Position TypedDict
inputs from system_params; returns SizingDecision / PyramidDecision / float / bool /
StepResult. NEVER mutates input position; NEVER reads filesystem / network / clock.

Public API:
  - calc_position_size(account, signal, atr, rvol, multiplier) -> SizingDecision   (SIZE-01..05)
  - get_trailing_stop(position, current_price, atr) -> float                        (EXIT-06/07)
  - check_stop_hit(position, high, low, atr) -> bool                                (EXIT-08/09)
  - check_pyramid(position, current_price, atr_entry) -> PyramidDecision            (PYRA-01..05)
  - compute_unrealised_pnl(position, current_price, multiplier, cost_aud_open) -> float  (D-13)
  - step(position, bar, indicators, old_signal, new_signal, account, multiplier, cost_aud_open) -> StepResult  (D-10)

Architecture (hexagonal-lite, CLAUDE.md): pure math ONLY. No I/O, no network, no
clock reads, no imports of state_manager / notifier / dashboard / main / pandas /
numpy. Allowed imports: dataclasses, math, signal_engine (LONG/SHORT/FLAT only),
system_params (constants + Position TypedDict).
'''
import dataclasses
import math

from signal_engine import FLAT, LONG, SHORT
from system_params import (
  ADX_EXIT_GATE,
  AUDUSD_COST_AUD,
  AUDUSD_NOTIONAL,
  MAX_PYRAMID_LEVEL,
  Position,
  RISK_PCT_LONG,
  RISK_PCT_SHORT,
  SPI_COST_AUD,
  SPI_MULT,
  TRAIL_MULT_LONG,
  TRAIL_MULT_SHORT,
  VOL_SCALE_MAX,
  VOL_SCALE_MIN,
  VOL_SCALE_TARGET,
)


# =========================================================================
# Dataclasses (per D-09 — frozen+slots for immutability + minimal memory)
# =========================================================================

@dataclasses.dataclass(frozen=True, slots=True)
class SizingDecision:
  '''Output of calc_position_size. `warning` is None for normal sizing,
  set to a descriptive string when contracts==0 (SIZE-05).'''
  contracts: int
  warning: str | None = None


@dataclasses.dataclass(frozen=True, slots=True)
class PyramidDecision:
  '''Output of check_pyramid. add_contracts is 0 (no pyramid this call) or 1
  (advance one level). Per D-12, never returns 2 — gap days that cross both
  thresholds still cap at 1 because only the current level's trigger is checked.'''
  add_contracts: int
  new_level: int


@dataclasses.dataclass(frozen=True, slots=True)
class ClosedTrade:
  '''Output captured inside StepResult when step() closes a position.
  Phase 3 record_trade consumes this to append to trade_log + adjust account.'''
  direction: str          # 'LONG' or 'SHORT'
  entry_date: str
  exit_date: str
  entry_price: float
  exit_price: float
```
