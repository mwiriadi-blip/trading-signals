```python
def _close_position(
  position: Position,
  bar: dict,
  multiplier: float,
  cost_aud_open: float,
  exit_reason: str,
  *,
  exit_price: float | None = None,
) -> ClosedTrade:
  '''Build a ClosedTrade for an exit. D-13 close-half cost is deducted here
  from the gross — by convention cost_aud_open == instrument_cost_aud / 2
  (e.g. SPI: 6.0/2 = 3.0); full round-trip is open + close.

  B-5 exit_price kwarg (reviews-revision 2026-04-21): when None (default),
  use bar['close'] — preserves the previous behavior for flat-signal and
  reversal closes. When provided (e.g. step() passes the computed stop level
  for exit_reason='stop_hit'), use that price for realised_pnl computation.
  This gives a more realistic best-case PnL for stop-hit fixtures than
  always falling back to bar.close.

  Pitfall 2 still holds for true gap-through-stop fill price (the actual
  fill could be worse than the stop on a gap day). Phase 3 record_trade can
  refine this further with broker-fill data; B-5 just makes the signal
  layer's stop-fill assumption more conservative than bar.close.
  '''
  effective_exit_price = exit_price if exit_price is not None else bar['close']
  direction_mult = 1.0 if position['direction'] == 'LONG' else -1.0
  gross = direction_mult * (effective_exit_price - position['entry_price']) * position['n_contracts'] * multiplier
  close_cost = cost_aud_open * position['n_contracts']
  realised_pnl = gross - close_cost
  return ClosedTrade(
    direction=position['direction'],
    entry_date=position['entry_date'],
    exit_date=bar['date'],
    entry_price=position['entry_price'],
    exit_price=effective_exit_price,
    n_contracts=position['n_contracts'],
    realised_pnl=realised_pnl,
    exit_reason=exit_reason,
  )
```

```python
def step(
  position: Position | None,
  bar: dict,
  indicators: dict,
  old_signal: int,
  new_signal: int,
  account: float,
  multiplier: float,
  cost_aud_open: float,
) -> StepResult:
  '''D-10 orchestrator (signature finalized by D-17). Composes individual
  callables in EXIT-then-ENTRY order per RESEARCH §Pattern 4. NEVER mutates
  input position; D-16 peak/trough updates are made on a SHALLOW COPY of the
  input dict. New positions opened by step() are fresh dicts. All dataclass
  returns are frozen.

  Order:
    0. D-16 peak/trough update on a shallow copy: LONG bumps peak via
       max(peak or entry, bar.high); SHORT bumps trough via min(trough or
       entry, bar.low). Only this copy is mutated; the input is untouched.
    1. EXIT-05: indicators.adx < ADX_EXIT_GATE -> close (exit_reason='adx_exit')
    2. Stop hit -> close (exit_reason='stop_hit'; B-5: exit_price = computed
       stop level via get_trailing_stop, NOT bar.close — gives a more
       realistic stop-fill PnL than always using close)
    3. Signal transition (9-cell):
       LONG/SHORT -> FLAT: close (exit_reason='flat_signal'; exit_price=bar.close)
       LONG -> SHORT (mirror): close (exit_reason='reversal'; exit_price=bar.close) + open new (D-19: sized off INPUT account)
       none -> LONG/SHORT: open new (no exit)
       LONG -> LONG / SHORT -> SHORT / none -> FLAT: hold
    4. Pyramid: only if position OPEN at end of 1-3; calls check_pyramid (pure)
       THEN applies the add to position_after per D-18 (n_contracts +=
       pyramid_decision.add_contracts; pyramid_level = pyramid_decision.new_level)
    5. Unrealised PnL on FINAL position state (post-D-18 mutation; B-6 invariant)

  RESEARCH A2: on stop-hit or ADX-exit day, no new entry fires even if
  new_signal would have triggered one. Conservative default for unknowable
  intraday timing.

  RESEARCH A4: ADX_EXIT_GATE compared against indicators['adx'] (today's ADX,
  not lagged).

  D-19: reversal sizing uses the INPUT `account` value. Phase 2 does NOT
  mutate account; account-state evolution from realised PnL belongs to
  Phase 3 (record_trade). The reversed-leg sizing therefore is a documented
  approximation against pre-close account; the approximation is small for
  typical $100k account vs single-trade realised PnL deltas.

  old_signal is accepted for caller-side logging context; step() does not
  branch on it (the 9-cell logic uses position direction + new_signal only).
  '''
  del old_signal  # accepted for API symmetry / logging context; not used in branching
  warnings: list[str] = []
  closed_trade: ClosedTrade | None = None
  sizing_decision: SizingDecision | None = None
  pyramid_decision: PyramidDecision | None = None

  # Phase 0: D-16 — peak/trough update on a shallow copy (preserves "never
  # mutate input" invariant). The copy becomes our working position; if no
  # position is open, skip the update.
  current_position: Position | None = None
  if position is not None:
    current_position = dict(position)  # shallow copy; OK because Position fields are scalars
    if current_position['direction'] == 'LONG':
      anchor = current_position['peak_price'] if current_position['peak_price'] is not None else current_position['entry_price']
      if math.isfinite(bar['high']):
        current_position['peak_price'] = max(anchor, bar['high'])
    else:  # SHORT
      anchor = current_position['trough_price'] if current_position['trough_price'] is not None else current_position['entry_price']
      if math.isfinite(bar['low']):
        current_position['trough_price'] = min(anchor, bar['low'])

  # Phase 1: EXIT-05 + stop-hit (only if position open)
  if current_position is not None:
    adx = indicators['adx']
    if math.isfinite(adx) and adx < ADX_EXIT_GATE:
      closed_trade = _close_position(current_position, bar, multiplier, cost_aud_open, 'adx_exit')
      current_position = None
    elif check_stop_hit(current_position, high=bar['high'], low=bar['low'], atr=indicators['atr']):
      # B-5: stop-hit exit_price = computed stop level (not bar.close)
      stop_level = get_trailing_stop(current_position, current_price=bar['close'], atr=indicators['atr'])
      closed_trade = _close_position(
        current_position, bar, multiplier, cost_aud_open, 'stop_hit',
        exit_price=stop_level,
      )
      current_position = None

  # Phase 2: 9-cell signal transition (only if no forced exit)
  if closed_trade is None and current_position is not None:
    if new_signal == FLAT:
      closed_trade = _close_position(current_position, bar, multiplier, cost_aud_open, 'flat_signal')
      current_position = None
    elif (
      (current_position['direction'] == 'LONG' and new_signal == SHORT)
      or (current_position['direction'] == 'SHORT' and new_signal == LONG)
    ):
      closed_trade = _close_position(current_position, bar, multiplier, cost_aud_open, 'reversal')
      current_position = None

  # New entry: position now None AND new_signal LONG/SHORT, BUT skip if forced exit fired (A2)
  if current_position is None and new_signal in (LONG, SHORT):
    forced_exit = closed_trade is not None and closed_trade.exit_reason in ('adx_exit', 'stop_hit')
    if not forced_exit:
      # D-19: reversal sizing uses INPUT account (Phase 2 does not mutate account).
      sizing_decision = calc_position_size(
        account=account, signal=new_signal,
        atr=indicators['atr'], rvol=indicators['rvol'], multiplier=multiplier,
      )
      if sizing_decision.warning is not None:
        warnings.append(sizing_decision.warning)
      if sizing_decision.contracts > 0:
        current_position = {
          'direction': 'LONG' if new_signal == LONG else 'SHORT',
          'entry_price': bar['close'],
          'entry_date': bar['date'],
          'n_contracts': sizing_decision.contracts,
          'pyramid_level': 0,
          'peak_price': bar['high'] if new_signal == LONG else None,
          'trough_price': bar['low'] if new_signal == SHORT else None,
          'atr_entry': indicators['atr'],
        }

  # Phase 3: Pyramid (only if position OPEN at end of phases 1-2)
  # D-18: check_pyramid is pure; step() applies the add to position_after.
  if current_position is not None:
    pyramid_decision = check_pyramid(
      current_position, current_price=bar['close'], atr_entry=current_position['atr_entry'],
    )
    if pyramid_decision.add_contracts > 0:
      # D-18: apply the add to position_after.
      current_position = dict(current_position)  # fresh copy in case current_position was the D-16 copy
      current_position['n_contracts'] = current_position['n_contracts'] + pyramid_decision.add_contracts
      current_position['pyramid_level'] = pyramid_decision.new_level

  # Phase 4: Unrealised PnL on FINAL position state (B-6: post any entry/pyramid mutation)
  if current_position is None:
    unrealised_pnl = 0.0
  else:
    unrealised_pnl = compute_unrealised_pnl(
      current_position, current_price=bar['close'], multiplier=multiplier, cost_aud_open=cost_aud_open,
    )

  return StepResult(
    position=current_position,
    closed_trade=closed_trade,
    sizing_decision=sizing_decision,
    pyramid_decision=pyramid_decision,
    unrealised_pnl=unrealised_pnl,
    warnings=warnings,
  )
```

```python
def _close_position_dict(position, bar, multiplier, cost_aud_open, exit_reason, exit_price=None):
  '''Inline _close_position mirror (B-5: exit_price kwarg). When exit_price is
  None, defaults to bar['close']; for 'stop_hit' callers pass the computed
  stop level so realised_pnl reflects the more conservative stop-fill.'''
  effective_exit_price = exit_price if exit_price is not None else bar['close']
  dm = 1.0 if position['direction'] == 'LONG' else -1.0
  gross = dm * (effective_exit_price - position['entry_price']) * position['n_contracts'] * multiplier
  close_cost = cost_aud_open * position['n_contracts']
  return {
    'direction': position['direction'],
    'entry_date': position['entry_date'],
    'exit_date': bar['date'],
    'entry_price': position['entry_price'],
    'exit_price': effective_exit_price,
    'n_contracts': position['n_contracts'],
    'realised_pnl': gross - close_cost,
    'exit_reason': exit_reason,
  }


def _step(position, bar, indicators, old_signal, new_signal, account, multiplier, cost_aud_open):

  warnings_list = []
  closed_trade = None
  sizing_decision = None
  pyramid_decision = None

  # Phase 0: D-16 — peak/trough update on a shallow copy
  current = None
  if position is not None:
    current = dict(position)
    if current['direction'] == 'LONG':
      anchor = current['peak_price'] if current['peak_price'] is not None else current['entry_price']
      if math.isfinite(bar['high']):
        current['peak_price'] = max(anchor, bar['high'])
    else:
      anchor = current['trough_price'] if current['trough_price'] is not None else current['entry_price']
      if math.isfinite(bar['low']):
```

```python
def _hash_expected(expected):
  '''SHA256 of JSON-serialized expected dict (sort_keys for determinism).'''
  canonical = json.dumps(expected, sort_keys=True, separators=(',', ':'), allow_nan=False)
  return hashlib.sha256(canonical.encode('utf-8')).hexdigest()
```
