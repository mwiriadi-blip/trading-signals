```python
def _vol_scale(rvol: float) -> float:
  '''SIZE-03: clip(VOL_SCALE_TARGET / rvol, VOL_SCALE_MIN, VOL_SCALE_MAX).

  D-03: NaN rvol or rvol <= 1e-9 -> VOL_SCALE_MAX (2.0). Returning the ceiling on
  degenerate input is the sizing-friendly choice — undersize is reflected in
  n_contracts==0; oversize from a junk RVol would be more dangerous.
  '''
  if not math.isfinite(rvol) or rvol <= 1e-9:
    return VOL_SCALE_MAX
  return max(VOL_SCALE_MIN, min(VOL_SCALE_MAX, VOL_SCALE_TARGET / rvol))
```

```python
def calc_position_size(
  account: float,
  signal: int,
  atr: float,
  rvol: float,
  multiplier: float,
) -> SizingDecision:
  '''SIZE-01..05. Returns SizingDecision; contracts=0 carries a 'size=0:' warning.

  Formula:
    risk_pct  = RISK_PCT_LONG (0.01) if signal==LONG else RISK_PCT_SHORT (0.005)
    trail_mult = TRAIL_MULT_LONG (3.0) if signal==LONG else TRAIL_MULT_SHORT (2.0)
    vol_scale = clip(VOL_SCALE_TARGET / rvol, VOL_SCALE_MIN, VOL_SCALE_MAX); NaN/zero rvol -> VOL_SCALE_MAX (D-03)
    stop_dist = trail_mult * atr * multiplier  (in AUD)
    n_raw    = (account * risk_pct / stop_dist) * vol_scale
    n_contracts = int(n_raw)  -- truncating int(); SIZE-04 explicitly forbids max(1, ...)

  SIZE-05: when n_contracts == 0 the SizingDecision.warning is a 'size=0: ...' diagnostic.
  Caller (Phase 4 orchestrator) inspects warning and skips the trade.

  signal must be LONG (1) or SHORT (-1). FLAT (0) is a programming error here —
  callers should never size a FLAT position. ATR must be finite and positive.
  '''
  if signal == LONG:
    risk_pct = RISK_PCT_LONG
    trail_mult = TRAIL_MULT_LONG
  elif signal == SHORT:
    risk_pct = RISK_PCT_SHORT
    trail_mult = TRAIL_MULT_SHORT
  else:
    # FLAT (0) is a caller error; surface as size=0 with a clear warning.
    return SizingDecision(
      contracts=0,
      warning=f'size=0: signal={signal} is not LONG or SHORT — caller must not size FLAT',
    )
  vol_scale = _vol_scale(rvol)
  stop_dist = trail_mult * atr * multiplier
  if not math.isfinite(stop_dist) or stop_dist <= 0.0:
    return SizingDecision(
      contracts=0,
      warning=(
        f'size=0: stop_dist={stop_dist} is not positive-finite '
        f'(atr={atr}, trail_mult={trail_mult}, multiplier={multiplier})'
      ),
    )
  n_raw = (account * risk_pct / stop_dist) * vol_scale
  n_contracts = int(n_raw)  # truncating; explicitly NOT max(1, ...)
  warning: str | None = None
  if n_contracts == 0:
    warning = (
      f'size=0: account={account:.2f}, atr={atr:.4f}, rvol={rvol:.4f}, '
      f'vol_scale={vol_scale:.4f}, stop_dist={stop_dist:.4f}, n_raw={n_raw:.6f}'
    )
  return SizingDecision(contracts=n_contracts, warning=warning)
```

```python
def compute_unrealised_pnl(
  position: Position,
  current_price: float,
  multiplier: float,
  cost_aud_open: float,
) -> float:
  '''D-13 split-cost unrealised mark-to-market PnL.

  formula:
    direction_mult = +1 if LONG else -1
    gross = direction_mult * (current_price - entry_price) * n_contracts * multiplier
    open_cost = cost_aud_open * n_contracts
    return gross - open_cost

  Pitfall 7: the direction_mult flips so SHORT positions show positive PnL when
  current_price < entry_price (losing-short shows negative PnL).

  cost_aud_open is supplied explicitly by the caller (Phase 4 orchestrator):
    SPI_COST_AUD / 2 = 3.0 per contract for SPI 200 mini
    AUDUSD_COST_AUD / 2 = 2.5 per contract for AUD/USD
  Phase 3 record_trade applies the matching close-cost-half on exit.

  No NaN guards on prices — if current_price or entry_price is NaN the result
  propagates NaN, which is the correct mark-to-market signal that data is bad.
  '''
  direction_mult = 1.0 if position['direction'] == 'LONG' else -1.0
  gross = direction_mult * (current_price - position['entry_price']) * position['n_contracts'] * multiplier
  open_cost = cost_aud_open * position['n_contracts']
  return gross - open_cost
```

```python
    SHORT same atr/mult: stop_dist=200, n_raw=(100000*0.005/200)*0.8=2.0, int=2.
    Both produce 2 — the ratio is clean and verifiable.

    assert long_d.contracts == 2, long_d
    assert short_d.contracts == 2, short_d
    # Differential: LONG at smaller atr produces more contracts than SHORT (3.0/2.0 ratio)
    long_smaller = calc_position_size(100000.0, LONG, 53.0, 0.15, 5.0)   # 1
    short_smaller = calc_position_size(100000.0, SHORT, 53.0, 0.15, 5.0)  # 0
    assert long_smaller.contracts > short_smaller.contracts, (long_smaller, short_smaller)
```
