    #!/usr/bin/env bash
    # deploy.sh — idempotent deploy script for trading-signals droplet.
    # Phase 11 D-20..D-25 (CONTEXT.md):
    #   D-20: strict mode (set -euo pipefail)
    #   D-21: runs as `trader` (sudoers grants passwordless restart for two units)
    #   D-22: branch safety check (must be on main)
    #   D-23: 8-step deploy sequence
    #   D-24: idempotent on no-op re-run (SC-3 verified on droplet)
    #   D-25: NO auto-rollback (fail-loud, operator intervenes)
    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    REPO_DIR="${SCRIPT_DIR}"
    cd "${REPO_DIR}"

    echo "[deploy] starting deploy at $(date '+%Y-%m-%d %H:%M:%S')"

    # D-22: branch safety check FIRST — refuse to deploy from non-main
    BRANCH=$(git rev-parse --abbrev-ref HEAD)
    if [ "${BRANCH}" != 'main' ]; then
      echo "[deploy] ERROR: expected branch 'main', got '${BRANCH}'. Aborting." >&2
      exit 1
    fi
    echo "[deploy] branch: ${BRANCH} — OK"

    # D-23 step 2: fetch
    echo "[deploy] fetching from origin..."
    git fetch origin main

    # D-23 step 3: fast-forward only — non-fast-forward exits non-zero via set -e
    echo "[deploy] pulling (ff-only)..."
    git pull --ff-only origin main

    # D-23 step 4: pip upgrade (silent on no-change)
    echo "[deploy] upgrading pip..."
    .venv/bin/pip install --upgrade pip --quiet

    # D-23 step 5: install requirements (idempotent — pip prints 'already satisfied' on no-change)
    echo "[deploy] installing requirements..."
    .venv/bin/pip install -r requirements.txt

    # D-23 step 6: restart BOTH units in one sudo invocation (sudoers lists both)
    echo "[deploy] restarting services..."
    sudo systemctl restart trading-signals trading-signals-web

    # Wait briefly for uvicorn to bind port 8000 (typical startup < 500ms; 3s is generous)
    sleep 3

    # D-23 step 7: smoke test — curl exits non-zero if status != 2xx or timeout exceeded
    echo "[deploy] smoke testing /healthz..."
    if ! curl -fsS --max-time 5 http://127.0.0.1:8000/healthz > /dev/null; then
      echo "[deploy] ERROR: /healthz smoke test failed. Check 'journalctl -u trading-signals-web -n 50'." >&2
      exit 1
    fi
    echo "[deploy] /healthz: OK"

    # D-23 step 8: success — echo final commit short-hash for the operator log
    COMMIT=$(git rev-parse --short HEAD)
    echo "[deploy] deploy complete. commit=${COMMIT}"
    ```

    '''Phase 11 INFRA-04 + D-20..D-25 — deploy.sh invariants guard.

    Parses deploy.sh and asserts every locked decision:
      D-20: shebang + set -euo pipefail
      D-22: branch safety check is the FIRST executable step after cd
      D-23: 8-step deploy sequence in correct order
        1. branch check
        2. git fetch origin main
        3. git pull --ff-only origin main
        4. pip install --upgrade pip
        5. pip install -r requirements.txt
        6. sudo systemctl restart trading-signals trading-signals-web
        7. curl -fsS --max-time 5 http://127.0.0.1:8000/healthz
        8. git rev-parse --short HEAD (commit echo)
      D-25: NO auto-rollback (no `git revert`, no `git reset --hard`)

    Plus structural checks:
      - script is executable (chmod +x)
      - bash -n deploy.sh exits 0 (syntax check)
      - smoke-test URL uses 127.0.0.1 (not 0.0.0.0, not localhost)
      - errors go to stderr (>&2)
      - both `exit 1` paths present (branch check + smoke test fail)
    '''
    import re
    import shutil
    import stat
    import subprocess
    from pathlib import Path

    import pytest

    DEPLOY_SH = Path('deploy.sh')


    @pytest.fixture(scope='module')
    def deploy_text() -> str:
      assert DEPLOY_SH.exists(), f'deploy.sh missing: {DEPLOY_SH}'
      return DEPLOY_SH.read_text()


    @pytest.fixture(scope='module')
    def deploy_lines(deploy_text: str) -> list:
      return deploy_text.splitlines()


    def _line_index(lines: list, pattern: str) -> int:
      '''Return 0-based index of the first line matching the regex pattern.'''
      regex = re.compile(pattern)
      for i, line in enumerate(lines):
        if regex.search(line):
          return i
      raise AssertionError(f'no line matched pattern: {pattern!r}')


    class TestDeployShStructure:
      '''D-20: shebang + strict mode; structural posture.'''

      def test_shebang_first_line(self, deploy_lines):
        assert deploy_lines[0] == '#!/usr/bin/env bash'

      def test_set_euo_pipefail_present(self, deploy_text):
        assert re.search(r'^set -euo pipefail$', deploy_text, re.MULTILINE)

      def test_file_is_executable(self):
        st = DEPLOY_SH.stat()
        assert st.st_mode & stat.S_IXUSR, f'mode={oct(st.st_mode)}'

      def test_bash_syntax_check_passes(self):
        bash = shutil.which('bash')
        if bash is None:
          pytest.skip('bash not available')
        result = subprocess.run([bash, '-n', str(DEPLOY_SH)], capture_output=True, text=True)
        assert result.returncode == 0, f'stderr={result.stderr!r}'

      def test_log_prefix_on_echoes(self, deploy_text):
        echo_lines = re.findall(r'^echo .+$', deploy_text, re.MULTILINE)
        assert len(echo_lines) >= 8, echo_lines
        prefixed = [e for e in echo_lines if '[deploy]' in e]
        assert len(prefixed) >= 8, prefixed


    class TestDeployShBranchSafety:
      '''D-22: branch check FIRST, refuse non-main.'''

      def test_branch_check_present(self, deploy_text):
        assert 'git rev-parse --abbrev-ref HEAD' in deploy_text

      def test_branch_compared_to_main(self, deploy_text):
        assert re.search(r'BRANCH.*!=.*main', deploy_text)

      def test_branch_check_exits_on_mismatch(self, deploy_text):
        assert "expected branch 'main'" in deploy_text
        assert 'exit 1' in deploy_text

      def test_branch_check_runs_before_fetch(self, deploy_lines):
        b = _line_index(deploy_lines, r'git rev-parse --abbrev-ref HEAD')
        f = _line_index(deploy_lines, r'git fetch origin main')
        assert b < f, (b, f)


    class TestDeployShSequence:
      '''D-23: 8-step deploy sequence — presence + cross-step ordering.'''

      def test_step_2_fetch_present(self, deploy_text):
        assert re.search(r'^git fetch origin main$', deploy_text, re.MULTILINE)

      def test_step_3_pull_ff_only_present(self, deploy_text):
        assert re.search(r'^git pull --ff-only origin main$', deploy_text, re.MULTILINE)

      def test_step_4_pip_upgrade_present(self, deploy_text):
        assert re.search(r'\.venv/bin/pip install --upgrade pip', deploy_text)

      def test_step_5_pip_install_requirements_present(self, deploy_text):
        assert re.search(r'\.venv/bin/pip install -r requirements\.txt', deploy_text)

      def test_step_6_systemctl_restart_both_units_in_one_command(self, deploy_text):
        '''Both unit names in one sudo invocation (sudoers entry lists both).'''
        assert 'sudo systemctl restart trading-signals trading-signals-web' in deploy_text

      def test_step_7_smoke_test_curl_exact(self, deploy_text):
        assert 'curl -fsS --max-time 5 http://127.0.0.1:8000/healthz' in deploy_text

      def test_step_8_commit_hash_echoed(self, deploy_text):
        assert 'git rev-parse --short HEAD' in deploy_text
        assert 'deploy complete' in deploy_text

      def test_order_fetch_before_pull(self, deploy_lines):
        f = _line_index(deploy_lines, r'git fetch origin main')
        p = _line_index(deploy_lines, r'git pull --ff-only origin main')
        assert f < p, (f, p)

      def test_order_pull_before_pip(self, deploy_lines):
        p = _line_index(deploy_lines, r'git pull --ff-only origin main')
        i = _line_index(deploy_lines, r'\.venv/bin/pip install -r requirements\.txt')
        assert p < i, (p, i)

      def test_order_pip_before_systemctl(self, deploy_lines):
        i = _line_index(deploy_lines, r'\.venv/bin/pip install -r requirements\.txt')
        s = _line_index(deploy_lines, r'sudo systemctl restart trading-signals trading-signals-web')
        assert i < s, (i, s)

      def test_order_systemctl_before_curl(self, deploy_lines):
        s = _line_index(deploy_lines, r'sudo systemctl restart trading-signals trading-signals-web')
        c = _line_index(deploy_lines, r'curl -fsS --max-time 5 http://127\.0\.0\.1:8000/healthz')
        assert s < c, (s, c)

      def test_order_curl_before_commit_echo(self, deploy_lines):
        c = _line_index(deploy_lines, r'curl -fsS --max-time 5 http://127\.0\.0\.1:8000/healthz')
        h = _line_index(deploy_lines, r'git rev-parse --short HEAD')
        assert c < h, (c, h)


    class TestDeployShSafety:
      '''D-25: no auto-rollback. Plus loopback-only smoke test, stderr errors.'''

      def test_no_git_revert(self, deploy_text):
        assert 'git revert' not in deploy_text

      def test_no_git_reset_hard(self, deploy_text):
        assert 'git reset --hard' not in deploy_text

      def test_no_rollback_keyword(self, deploy_text):
        assert re.search(r'\brollback\b', deploy_text, re.IGNORECASE) is None

        assert '0.0.0.0' not in deploy_text
        urls = re.findall(r'http://[^\s/]+', deploy_text)
        for url in urls:
          assert '127.0.0.1' in url, f'non-loopback URL: {url}'

      def test_errors_go_to_stderr(self, deploy_text):
        assert deploy_text.count('>&2') >= 2

      def test_two_exit_one_paths(self, deploy_text):
        exit_lines = [l for l in deploy_text.splitlines() if l.strip() == 'exit 1']

      def test_no_daemon_reload(self, deploy_text):
        '''Unit file is installed once; deploy reuses it. No reload needed.'''
        assert 'daemon-reload' not in deploy_text
    ```
