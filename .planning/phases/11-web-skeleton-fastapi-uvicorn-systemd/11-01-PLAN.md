    ```python
    '''GET /healthz — Phase 11 WEB-07 + D-13..D-19.

    Liveness check endpoint. Always returns HTTP 200 with the schema:
      {'status': 'ok', 'last_run': <YYYY-MM-DD str|null>, 'stale': <bool>}

    Contract (CONTEXT.md):
      D-13: response body is fixed JSON schema above
      D-14: always HTTP 200 if the FastAPI process is alive (process-liveness
            is the contract; pipeline health is in the body via `stale`)
      D-15: last_run from state_manager.load_state() — local import (C-2)
      D-16: stale=True iff last_run is more than 2 days older than now AWST
            (Phase 8 ERR-05 threshold reused). last_run=None never sets stale.
      D-17: exempt from auth (Phase 13 middleware must skip /healthz)
      D-18: no caching — each request calls load_state() fresh
      D-19: NEVER non-200; on any exception, return degraded but still 200
            and log WARN with [Web] prefix.

    Hex boundary: state_manager import is LOCAL (inside the function body)
    per C-2 pattern from main.py::_render_dashboard_never_crash. Module-top
    imports are stdlib + fastapi only.
    '''
    import logging

    from fastapi import FastAPI

    logger = logging.getLogger(__name__)


    def register(app: FastAPI) -> None:
      '''Register GET /healthz on the given FastAPI instance.

      Called by web.app.create_app(). Phase 13+ patterns will add
      register() functions for state.py, trades.py, dashboard.py.
      '''

      @app.get('/healthz')
      def healthz() -> dict:
        '''GET /healthz handler — see module docstring for D-13..D-19 contract.

        Returns a plain dict; FastAPI serialises to JSON automatically and
        sets Content-Type: application/json.
        '''
        # C-2 local imports: import-time errors in state_manager are caught
        # by the same except Exception that catches runtime load_state errors.
        import zoneinfo
        from datetime import date as _date
        from datetime import datetime

        try:
          from state_manager import load_state  # local import — C-2 hex pattern

          state = load_state()
          last_run = state.get('last_run')  # 'YYYY-MM-DD' str or None (D-15)

          stale = False
          if last_run is not None:
            # D-16: stale if more than 2 days older than now AWST
            awst = zoneinfo.ZoneInfo('Australia/Perth')
            now_awst = datetime.now(awst)
            try:
              last_dt = _date.fromisoformat(last_run)
              delta_days = (now_awst.date() - last_dt).days
              stale = delta_days > 2
            except (TypeError, ValueError):
              # Malformed last_run — treat as non-stale rather than crash
              stale = False

          return {'status': 'ok', 'last_run': last_run, 'stale': stale}

        except Exception as exc:  # noqa: BLE001 — D-19 never-crash posture
          logger.warning(
            '[Web] /healthz load_state failed: %s: %s',
            type(exc).__name__,
            exc,
          )
          return {'status': 'ok', 'last_run': None, 'stale': False}
    ```
