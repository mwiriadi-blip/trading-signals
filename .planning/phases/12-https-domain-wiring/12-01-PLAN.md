    #
    # On first `certbot --nginx -d signals.<owned-domain>.com` run, certbot injects:
    #   (a) ssl_certificate + ssl_certificate_key lines (marked `# managed by Certbot`)
    #   (b) a new `server { listen 80; ... return 301 https://$host$request_uri; }`
    #   (c) a temporary location for /.well-known/acme-challenge/ during renewal

    # timer — code side) and WEB-04 (HSTS exact value, no preload per D-12).

    # D-10: rate-limit zone at http{} scope. nginx inherits `limit_req` into
    # nested locations ONLY when no child-level `limit_req` exists — so the
    # /.well-known/acme-challenge/ block below disables rate-limit for ACME
    # challenges (T-12-02). [CITED: nginx limit_req module docs]
    limit_req_zone $binary_remote_addr zone=healthz:10m rate=10r/m;

    server {
      listen 443 ssl;
      listen [::]:443 ssl;
      http2 on;

      server_name signals.<owned-domain>.com;

      # --- TLS tuning — Mozilla Intermediate profile (2024 rev) ---
      # [CITED: https://ssl-config.mozilla.org/ — Mozilla SSL Configuration Generator]
      ssl_protocols TLSv1.2 TLSv1.3;
      ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384:DHE-RSA-CHACHA20-POLY1305;
      ssl_prefer_server_ciphers off;
      ssl_session_cache shared:SSL:10m;
      ssl_session_timeout 1d;
      ssl_session_tickets off;

      # --- OCSP stapling ---
      # certbot injects `ssl_trusted_certificate` post-issuance; until then
      # nginx silently fail-softs stapling (no crash).
      ssl_stapling on;
      ssl_stapling_verify on;
      resolver 1.1.1.1 8.8.8.8 valid=300s;
      resolver_timeout 5s;

      # NOTE: ssl_certificate + ssl_certificate_key lines are NOT in this file.
      # Certbot injects them on first run; committing them here creates
      # non-idempotent re-runs and leaks cert paths into git history.

      # --- Security headers (D-11 — ALL at server scope, ALL `always`) ---
      # `always` flag ensures emission on 4xx/5xx too (RESEARCH Pitfall 2).
      # Server scope is mandatory — `add_header` inside a location REPLACES
      # (not extends) parent-scope headers (RESEARCH Pitfall 3). If Phase 13+
      # needs per-route headers, ALL security headers must be redeclared in
      # that location block, OR use the third-party headers-more-nginx-module.
      add_header Strict-Transport-Security 'max-age=31536000; includeSubDomains' always;
      add_header X-Content-Type-Options 'nosniff' always;
      add_header X-Frame-Options 'DENY' always;
      add_header Referrer-Policy 'strict-origin-when-cross-origin' always;

      # --- Let's Encrypt ACME challenge carve-out (D-09, T-12-02) ---
      # Nested location with NO limit_req disables rate-limit for ACME.
      # certbot's --nginx plugin serves files from a temp webroot during renewal.
      location /.well-known/acme-challenge/ {
        try_files $uri =404;
      }

      # --- /healthz — rate-limited (D-10) ---
      location = /healthz {
        limit_req zone=healthz burst=10 nodelay;
        limit_req_status 429;

        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 5s;
        proxy_connect_timeout 2s;
      }

      # --- Everything else — no rate limit; Phase 13 adds auth ---
      location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 30s;
        proxy_connect_timeout 5s;
      }
    }
