    ```python
    """Phase 28 / DEBT-01 / UAT-17-3: Trace-panel toggle cookie persists across one reload.

    Sourced from .planning/milestones/v1.2-phases/17-.../17-VERIFICATION.md UAT-3.

    What is being verified together (per 28-CONTEXT.md specifics):
      1. Cookie WRITE — clicking the trace-panel toggle sets a cookie.
      2. Cookie READ — page.reload() restores the toggle to its post-click state.
      3. No session loss — user remains logged in across the reload.

    A spec that proves only (1) misses read regressions; only (2) misses
    write regressions. The whole flow is the contract.
    """
    from __future__ import annotations

    import os

    import pytest

    pytestmark = pytest.mark.uat

    DASHBOARD_PATH = os.environ.get("UAT_17_DASHBOARD_PATH", "/markets/SPI200/dashboard")

    # Selector contract — adjust if production differs and record deviation
    # in 28-03-SUMMARY.md. Do not loosen the assertion shape.
    TOGGLE_SELECTOR = "[data-trace-toggle]"
    PANEL_SELECTOR = "[data-trace-panel]"
    COLLAPSED_ATTR = "data-collapsed"  # 'true' | 'false'
    LOGGED_IN_INDICATOR = "[data-user-menu]"  # present iff session active


    def _toggle_state(page) -> str:
        """Return 'true' or 'false' for the panel's collapsed attribute."""
        return page.locator(PANEL_SELECTOR).get_attribute(COLLAPSED_ATTR) or "false"


    def test_trace_panel_toggle_persists_across_reload(page, base_url):
        page.goto(f"{base_url}{DASHBOARD_PATH}")
        page.wait_for_selector(LOGGED_IN_INDICATOR, timeout=15_000)
        page.wait_for_selector(PANEL_SELECTOR, timeout=15_000)

        initial = _toggle_state(page)

        # ACT: click toggle (cookie WRITE).
        page.locator(TOGGLE_SELECTOR).click()
        page.wait_for_function(
            "(args) => {"
            "  const el = document.querySelector(args.sel);"
            "  return el &amp;&amp; el.getAttribute(args.attr) !== args.was;"
            "}",
            arg={"sel": PANEL_SELECTOR, "attr": COLLAPSED_ATTR, "was": initial},
            timeout=5_000,
        )
        post_click = _toggle_state(page)
        assert post_click != initial, (
            f"Toggle click did not flip state: initial={initial} post_click={post_click}"
        )

        # ASSERT cookie was actually written (not just JS state).
        cookies = page.context.cookies()
        toggle_cookie_names = [c["name"] for c in cookies if "trace" in c["name"].lower()]
        assert toggle_cookie_names, (
            f"No trace-panel cookie written after toggle. "
            f"Cookie names present: {[c['name'] for c in cookies]}"
        )

        # ACT: reload (cookie READ).
        page.reload()
        page.wait_for_selector(PANEL_SELECTOR, timeout=15_000)

        # ASSERT 1: visual state preserved.
        post_reload = _toggle_state(page)
        assert post_reload == post_click, (
            f"Toggle state lost across reload: post_click={post_click} post_reload={post_reload}"
        )

        # ASSERT 2: still logged in (no session loss).
        assert page.locator(LOGGED_IN_INDICATOR).count() == 1, (
            "User-menu indicator missing after reload — session was lost."
        )
    ```

    Notes:
    - Selectors are the contract; if production uses different attributes (e.g., `aria-expanded` instead of `data-collapsed`), update here and document in 28-03-SUMMARY.md.
    - The cookie-name pattern `"trace" in name.lower()` is intentionally loose because the exact cookie key is implementation detail; the assertion is "some cookie related to trace was set".
