    class TestFcntlLock:
      '''Phase 14 D-13: cross-process state.json coordination via fcntl.LOCK_EX.

      Lock is on the destination state.json. Acquired in _atomic_write before
      tempfile→replace; released by explicit LOCK_UN + os.close. Blocking-
      indefinite per D-13.

      Cross-process correctness verified via multiprocessing.Process holder
      that opens state.json + flock(LOCK_EX) + sleeps 0.5s; main test calls
      save_state() and asserts elapsed >= 0.4s (proves contention) and < 1.5s
      (proves no zombie wait).

      Per CONTEXT D-13 regression discipline: existing TestAtomicity green;
      lock acquisition latency < 100ms in normal operation.
      '''

      @staticmethod
      def _hold_lock_for(path_str, hold_seconds, ready_event):
        '''Subprocess body: open state.json, acquire LOCK_EX, signal ready,
        sleep, release on exit. Module-level so multiprocessing can pickle.'''
        import fcntl
        import os
        import time
        fd = os.open(path_str, os.O_RDWR | os.O_CREAT, 0o644)
        try:
          fcntl.flock(fd, fcntl.LOCK_EX)
          ready_event.set()
          time.sleep(hold_seconds)
        finally:
          fcntl.flock(fd, fcntl.LOCK_UN)
          os.close(fd)

      def test_save_state_blocks_when_external_lock_held(self, tmp_path) -> None:
        '''D-13: save_state blocks while another process holds LOCK_EX.

        Lower bound (>= 0.4s) proves contention; upper bound (< 1.5s) proves
        the lock is released and acquired without zombie waits.'''
        import multiprocessing as mp
        import time
        path = tmp_path / 'state.json'
        save_state(reset_state(), path=path)  # seed valid JSON

        ctx = mp.get_context('spawn')  # macOS-safe; avoids fork+pytest interaction
        ready = ctx.Event()
        hold_s = 0.5
        proc = ctx.Process(
          target=self._hold_lock_for,
          args=(str(path), hold_s, ready),
        )
        proc.start()
        try:
          assert ready.wait(timeout=2.0), 'subprocess did not acquire lock in 2s'
          state = reset_state()
          state['account'] = 12345.0  # marker
          start = time.perf_counter()
          save_state(state, path=path)
          elapsed = time.perf_counter() - start
        finally:
          proc.join(timeout=2.0)
          if proc.is_alive():
            proc.terminate()
            proc.join()
        assert elapsed >= 0.4, (
          f'D-13: save_state did not block on contended lock; elapsed={elapsed:.3f}s'
        )
        assert elapsed < 1.5, (
          f'D-13: save_state took too long after lock release; elapsed={elapsed:.3f}s'
        )
        assert load_state(path=path)['account'] == 12345.0
